---
title: "List Sessions"
api: "GET /api/v1/sessions"
description: "Retrieve a paginated list of Claude Code sessions, optionally filtered by project directory."
---

## Query Parameters

<ParamField query="limit" type="integer" default="20">
  Maximum number of sessions to return.
</ParamField>

<ParamField query="offset" type="integer" default="0">
  Number of sessions to skip for pagination.
</ParamField>

<ParamField query="project" type="string">
  Filter sessions by project directory path. Only sessions created within this directory are returned.
</ParamField>

## Response

<ResponseField name="sessions" type="array" required>
  Array of session summary objects.
  <Expandable title="Session object">
    <ResponseField name="sessionId" type="string">
      Unique session identifier.
    </ResponseField>
    <ResponseField name="createdAt" type="string">
      ISO 8601 timestamp of session creation.
    </ResponseField>
    <ResponseField name="lastModified" type="string">
      ISO 8601 timestamp of last activity.
    </ResponseField>
    <ResponseField name="status" type="string">
      Session status: `completed`, `running`, `errored`, `aborted`.
    </ResponseField>
    <ResponseField name="firstPrompt" type="string">
      The initial prompt that launched the session.
    </ResponseField>
    <ResponseField name="summary" type="string">
      Auto-generated summary of the session.
    </ResponseField>
    <ResponseField name="cwd" type="string">
      Working directory where the session ran.
    </ResponseField>
    <ResponseField name="tag" type="string">
      Optional user-assigned tag.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="total" type="integer" required>
  Total number of sessions matching the filter.
</ResponseField>

## Examples

<CodeGroup>

```bash curl
curl http://127.0.0.1:3000/api/v1/sessions?limit=5&offset=0
```

```bash CLI
ccm sessions list --limit 5
```

```bash CLI with project filter
ccm sessions list --project /Users/dev/my-project
```

</CodeGroup>
