---
title: "List Sessions"
api: "GET /api/v1/sessions"
description: "Retrieve the enriched session catalog, combining Claude filesystem discovery with indexed metadata, lineage, and directory context."
---

## Query Parameters

<ParamField query="limit" type="integer" default="20">
  Maximum number of catalog sessions to return.
</ParamField>

<ParamField query="offset" type="integer" default="0">
  Number of catalog sessions to skip.
</ParamField>

<ParamField query="project" type="string">
  Filter sessions by project directory path.
</ParamField>

<ParamField query="metadataKey" type="string">
  Filter sessions by a registered metadata key.
</ParamField>

<ParamField query="metadataValue" type="string">
  Filter sessions by a metadata value substring.
</ParamField>

<ParamField query="lineage" type="string">
  Filter by relationship summary: `all`, `standalone`, `subagent`, or `team`.
</ParamField>

<ParamField query="team" type="string">
  Restrict results to sessions associated with a team name.
</ParamField>

## Response

<ResponseField name="sessions" type="array" required>
  Array of enriched catalog session objects.
  <Expandable title="Catalog session object">
    <ResponseField name="sessionId" type="string">
      Unique Claude session identifier.
    </ResponseField>
    <ResponseField name="id" type="string">
      Convenience alias for `sessionId`.
    </ResponseField>
    <ResponseField name="createdAt" type="number">
      Unix timestamp in milliseconds.
    </ResponseField>
    <ResponseField name="lastModified" type="number">
      Unix timestamp in milliseconds.
    </ResponseField>
    <ResponseField name="project" type="string">
      Basename of the session working directory.
    </ResponseField>
    <ResponseField name="directoryPath" type="string">
      Exact working directory path used for grouping.
    </ResponseField>
    <ResponseField name="indexed" type="boolean">
      Whether the session is present in the SQLite index.
    </ResponseField>
    <ResponseField name="messageCount" type="integer">
      Indexed message count when known.
    </ResponseField>
    <ResponseField name="firstPrompt" type="string">
      The initial prompt that launched the session.
    </ResponseField>
    <ResponseField name="summary" type="string">
      Auto-generated summary of the session.
    </ResponseField>
    <ResponseField name="cwd" type="string">
      Working directory where the session ran.
    </ResponseField>
    <ResponseField name="metadata" type="array">
      Registered session metadata entries attached to the session.
    </ResponseField>
    <ResponseField name="lineage" type="object">
      Relationship summary including subagent counts and inferred team membership.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="total" type="integer" required>
  Total number of catalog sessions matching the filters.
</ResponseField>

## Examples

<CodeGroup>

```bash curl
curl "http://127.0.0.1:3000/api/v1/sessions?limit=5&offset=0"
```

```bash team-filtered catalog
curl "http://127.0.0.1:3000/api/v1/sessions?lineage=team&team=delivery"
```

```bash CLI
ccm sessions list --limit 5
```

</CodeGroup>
