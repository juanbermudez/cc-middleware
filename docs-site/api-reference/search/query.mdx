---
title: "Search Sessions"
api: "GET /api/v1/search"
description: "Full-text search across indexed sessions and messages with filters for metadata, lineage, and team context."
---

## Query Parameters

<ParamField query="q" type="string" default="">
  Search query string. Searches indexed session summaries, prompts, messages, and searchable session metadata.
</ParamField>

<ParamField query="project" type="string">
  Filter results to a specific indexed project name.
</ParamField>

<ParamField query="dateFrom" type="number">
  Unix timestamp (milliseconds). Only return sessions modified after this time.
</ParamField>

<ParamField query="dateTo" type="number">
  Unix timestamp (milliseconds). Only return sessions modified before this time.
</ParamField>

<ParamField query="tags" type="string">
  Comma-separated list of tags to filter by.
</ParamField>

<ParamField query="status" type="string">
  Filter by indexed session status.
</ParamField>

<ParamField query="metadataKey" type="string">
  Restrict results to sessions with a matching metadata key.
</ParamField>

<ParamField query="metadataValue" type="string">
  Restrict results to sessions whose filterable metadata values include this substring.
</ParamField>

<ParamField query="lineage" type="string">
  Filter results to `all`, `standalone`, `subagent`, or `team`.
</ParamField>

<ParamField query="team" type="string">
  Restrict results to sessions associated with a specific team name.
</ParamField>

<ParamField query="limit" type="integer" default="20">
  Maximum number of results.
</ParamField>

<ParamField query="offset" type="integer" default="0">
  Number of results to skip.
</ParamField>

## Response

<ResponseField name="sessions" type="array" required>
  Array of indexed search result objects.
  <Expandable title="Search result object">
    <ResponseField name="id" type="string">
      Matching session ID.
    </ResponseField>
    <ResponseField name="summary" type="string">
      Indexed session summary.
    </ResponseField>
    <ResponseField name="relevanceScore" type="number">
      Relevance score for the indexed session match.
    </ResponseField>
    <ResponseField name="matchHighlights" type="array">
      Match snippets from indexed content or searchable metadata.
    </ResponseField>
    <ResponseField name="metadata" type="array">
      Session metadata entries included with the search result.
    </ResponseField>
    <ResponseField name="lineage" type="object">
      Subagent/team relationship summary for the root session.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="total" type="integer" required>
  Total number of matching indexed sessions.
</ResponseField>

<ResponseField name="queryTimeMs" type="integer" required>
  End-to-end query duration in milliseconds.
</ResponseField>

## Examples

<CodeGroup>

```bash curl (basic search)
curl "http://127.0.0.1:3000/api/v1/search?q=error+handling"
```

```bash curl (metadata + lineage)
curl "http://127.0.0.1:3000/api/v1/search?q=&metadataKey=workflow&metadataValue=delivery&lineage=team&team=delivery"
```

```bash CLI
ccm sessions search "error handling"
```

</CodeGroup>
