## Task 6.5.1: Hooks + Session Integration

### What to Test

The SDK bridge (`createSDKHooks()`) generates hook callbacks that dispatch to our event bus. This test proves that when a real session runs, our hooks actually fire.

### Implementation: `tests/integration/hooks-session.test.ts`

```typescript
import { query } from '@anthropic-ai/claude-agent-sdk';
import { HookEventBus } from '../../src/hooks/event-bus.js';
import { BlockingHookRegistry } from '../../src/hooks/blocking.js';
import { createSDKHooks } from '../../src/hooks/sdk-bridge.js';

describe('Hooks + Session Integration', () => {
  it('should fire PreToolUse and PostToolUse events during a real session', async () => {
    const eventBus = new HookEventBus();
    const registry = new BlockingHookRegistry();

    const preToolEvents: any[] = [];
    const postToolEvents: any[] = [];
    eventBus.on('PreToolUse', (input) => preToolEvents.push(input));
    eventBus.on('PostToolUse', (input) => postToolEvents.push(input));

    const hooks = createSDKHooks(eventBus, registry);

    for await (const message of query({
      prompt: 'Read the file package.json and tell me the project name',
      options: {
        allowedTools: ['Read'],
        hooks,
        maxTurns: 3,
      }
    })) {
      // consume messages
    }

    expect(preToolEvents.length).toBeGreaterThan(0);
    expect(preToolEvents.some(e => e.tool_name === 'Read')).toBe(true);
    expect(postToolEvents.length).toBeGreaterThan(0);
    expect(postToolEvents.some(e => e.tool_name === 'Read')).toBe(true);
  }, 60000);

  it('should block a tool when blocking handler denies it', async () => {
    const eventBus = new HookEventBus();
    const registry = new BlockingHookRegistry();

    // Register a deny handler for Bash
    registry.register('PreToolUse', async (input) => ({
      hookSpecificOutput: {
        hookEventName: 'PreToolUse',
        permissionDecision: 'deny' as const,
        permissionDecisionReason: 'Blocked by integration test',
      }
    }), { matcher: 'Bash' });

    const hooks = createSDKHooks(eventBus, registry);

    let resultText = '';
    for await (const message of query({
      prompt: 'Run the command: echo "hello world"',
      options: {
        hooks,
        maxTurns: 3,
      }
    })) {
      if (message.type === 'result' && message.subtype === 'success') {
        resultText = message.result;
      }
    }

    // Claude should have been told the tool was blocked
    // The result should indicate the tool was denied (not contain "hello world" from actual execution)
    // This verifies the blocking hook actually prevented execution
  }, 60000);
});
```

### Verification
- PreToolUse("Read") event arrives on bus with real tool_input containing file_path
- PostToolUse("Read") event arrives with real tool_response
- Blocking deny prevents Bash execution

## Task 6.5.2: Permissions + Session Integration

### What to Test

The `createCanUseTool()` function connected to a real PolicyEngine should allow/deny tools during a real session.

### Implementation: `tests/integration/permissions-session.test.ts`

```typescript
import { query } from '@anthropic-ai/claude-agent-sdk';
import { PolicyEngine } from '../../src/permissions/policy.js';
import { createCanUseTool, PermissionManager } from '../../src/permissions/handler.js';
import { HookEventBus } from '../../src/hooks/event-bus.js';

describe('Permissions + Session Integration', () => {
  it('should allow Read but deny Bash based on policy', async () => {
    const policy = new PolicyEngine({
      rules: [
        { id: 'deny-bash', toolName: 'Bash', behavior: 'deny', priority: 1 },
        { id: 'allow-read', toolName: 'Read', behavior: 'allow', priority: 2 },
      ],
      defaultBehavior: 'deny',
    });
    const eventBus = new HookEventBus();
    const permissionManager = new PermissionManager();
    const canUseTool = createCanUseTool({ policyEngine: policy, eventBus, permissionManager });

    let resultText = '';
    for await (const message of query({
      prompt: 'First try to run "echo test" with Bash, then read the file package.json',
      options: {
        canUseTool,
        maxTurns: 5,
      }
    })) {
      if (message.type === 'result' && message.subtype === 'success') {
        resultText = message.result;
      }
    }

    // Result should mention reading package.json succeeded
    // Result should indicate Bash was denied
    expect(resultText).toBeTruthy();
  }, 60000);
});
```

### Verification
- Bash tool calls are denied (Claude gets deny message)
- Read tool calls proceed normally
- PolicyEngine decisions actually affect the session

## Task 6.5.3: HTTP Hook Server + Claude CLI Integration

### What to Test

Start our hook HTTP server, then run `claude -p` with settings that route hooks to our server. This proves the plugin integration path works with a real Claude Code session.

### Implementation: `tests/integration/hookserver-cli.test.ts`

```typescript
import { execSync } from 'child_process';
import { HookEventBus } from '../../src/hooks/event-bus.js';
import { BlockingHookRegistry } from '../../src/hooks/blocking.js';
import { createHookServer } from '../../src/hooks/server.js';

describe('HTTP Hook Server + Claude CLI', () => {
  it('should receive PostToolUse events from a real claude -p session', async () => {
    const eventBus = new HookEventBus();
    const registry = new BlockingHookRegistry();
    const port = 13580 + Math.floor(Math.random() * 1000);

    const server = await createHookServer({ port, host: '127.0.0.1', eventBus, blockingRegistry: registry });
    await server.start();

    const receivedEvents: any[] = [];
    eventBus.on('PostToolUse', (input) => receivedEvents.push(input));

    try {
      // Run claude -p with settings that route PostToolUse hooks to our server
      const settings = JSON.stringify({
        hooks: {
          PostToolUse: [{
            matcher: 'Read',
            hooks: [{ type: 'http', url: `http://127.0.0.1:${port}/hooks/PostToolUse`, timeout: 10 }]
          }]
        }
      });

      execSync(
        `claude -p "Read the file package.json and tell me the name field" --allowedTools "Read" --settings '${settings}' --output-format json`,
        { cwd: '/Users/zef/Desktop/cc-middleware', timeout: 60000, encoding: 'utf-8' }
      );

      // Give a moment for async HTTP delivery
      await new Promise(r => setTimeout(r, 1000));

      expect(receivedEvents.length).toBeGreaterThan(0);
      expect(receivedEvents[0].tool_name).toBe('Read');
    } finally {
      await server.stop();
    }
  }, 90000);
});
```

### Verification
- Hook server receives real PostToolUse event from `claude -p`
- Event has real tool_name, tool_input, session_id
- Proves the plugin HTTP hook path works end-to-end

```typescript
import { launchSession } from '../../src/sessions/launcher.js';

```

### Verification

- Rename persists and is readable

## Task 6.5.5: Streaming Abort

### What to Test

Launch a long-running streaming session and abort it mid-stream.

### Implementation: `tests/integration/streaming-abort.test.ts`

```typescript
import { SessionManager } from '../../src/sessions/manager.js';

describe('Streaming Abort', () => {
  it('should abort a streaming session mid-stream', async () => {
    const manager = new SessionManager();
    const events: string[] = [];

    manager.on('session:started', () => events.push('started'));
    manager.on('session:aborted', () => events.push('aborted'));

    const stream = await manager.launchStreaming({
      prompt: 'Write a very long essay about the history of computing, at least 2000 words',
      maxTurns: 1,
      includePartialMessages: true,
    });

    // Wait for first streaming events
    let receivedEvents = 0;
    for await (const event of stream.events) {
      receivedEvents++;
      if (receivedEvents >= 5) {
        // Abort after receiving some events
        await manager.abort(stream.sessionId);
        break;
      }
    }

    expect(events).toContain('started');
    expect(receivedEvents).toBeGreaterThanOrEqual(5);
    expect(manager.getActiveSessions()).toHaveLength(0);

    await manager.destroy();
  }, 60000);
});
```

### Verification
- Streaming session starts, events arrive
- Abort stops the session
- Session removed from active list
- No hanging processes

## Task 6.5.6: Full Middleware Smoke Test

### What to Test

Create the complete middleware stack (all systems wired together) and run a session through it, verifying the full event flow.

### Implementation: `tests/integration/smoke.test.ts`

```typescript
import { SessionManager } from '../../src/sessions/manager.js';
import { HookEventBus } from '../../src/hooks/event-bus.js';
import { BlockingHookRegistry } from '../../src/hooks/blocking.js';
import { createSDKHooks } from '../../src/hooks/sdk-bridge.js';
import { PolicyEngine } from '../../src/permissions/policy.js';
import { createCanUseTool, PermissionManager } from '../../src/permissions/handler.js';
import { AgentRegistry } from '../../src/agents/registry.js';

describe('Full Middleware Smoke Test', () => {
  it('should run a session through the complete middleware stack', async () => {
    // Build the stack
    const eventBus = new HookEventBus();
    const registry = new BlockingHookRegistry();
    const hooks = createSDKHooks(eventBus, registry);

    const policy = new PolicyEngine({
      rules: [{ id: 'allow-all', toolName: '.*', behavior: 'allow', priority: 10 }],
      defaultBehavior: 'allow',
    });
    const permissionManager = new PermissionManager();
    const canUseTool = createCanUseTool({ policyEngine: policy, eventBus, permissionManager });

    const agentRegistry = new AgentRegistry();
    const sessionManager = new SessionManager();

    // Track events
    const lifecycleEvents: string[] = [];
    const hookEvents: string[] = [];

    sessionManager.on('session:started', () => lifecycleEvents.push('started'));
    sessionManager.on('session:completed', () => lifecycleEvents.push('completed'));
    eventBus.on('*', (eventType) => hookEvents.push(eventType as string));

    // Launch a session through the manager with all systems active
    const result = await sessionManager.launch({
      prompt: 'What is 2+2? Reply with just the number.',
      maxTurns: 1,
      hooks,
      canUseTool,
    });

    // Verify full flow
    expect(result.sessionId).toBeTruthy();
    expect(result.result).toContain('4');
    expect(lifecycleEvents).toContain('started');
    expect(lifecycleEvents).toContain('completed');
    // Hook events should have fired (at minimum SessionStart, Stop)
    expect(hookEvents.length).toBeGreaterThan(0);

    await sessionManager.destroy();
  }, 60000);
});
```

### Verification
- All systems instantiate and wire together without errors
- Session launches through manager
- Lifecycle events fire (started, completed)
- Hook events fire through the event bus
- Permission evaluation runs (allow-all policy)
- Result is correct ("4")
- Clean shutdown
